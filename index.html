                    def Iterativefactorial(m):

                            fact = 1
                            for i in range(2, m+1):
                                    fact *= i
                            return fact

                    def Recursivefactorial(n):

                             if n == 0:
                                      return 1

                            return n * Recursivefactorial(n-1)


                    num = int(input("enter the number : \n"))
